    const div = document.createElement('div');

    div.className = 'g-cat';

    const items = cat.items.map((item,i)=>{

      const key = `${id}-${cat.title}-${i}`;

      const checked = checkStates[key] || false;

      return `<div class="g-item${checked?' done':''}" id="gi-${key}">

        <input type="checkbox" id="cb-${key}" ${checked?'checked':''} onchange="toggleGroc('${key}')">

        <label for="cb-${key}">${item}</label>

      </div>`;

    }).join('');

    div.innerHTML = `<h4>${cat.title}</h4>${items}`;

    grid.appendChild(div);

  });

}


function toggleGroc(key){

  checkStates[key] = !checkStates[key];

  const item = document.getElementById('gi-'+key);

  if(item) item.classList.toggle('done', checkStates[key]);

}
